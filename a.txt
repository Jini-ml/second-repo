from plus_minus import add_func, sub_func
from mul_div import mul_func, div_func

## 전역 변수부
num1, num2, res = 100, 200, 0

res = add_func(num1, num2)
print(num1, "+", num2, "=", res)

res = sub_func(num1, num2)
print(num1, "-", num2, "=", res)

res = mul_func(num1, num2)
print(num1, "*", num2, "=", res)

res = div_func(num1, num2)
print(num1, "/", num2, "=", res)
